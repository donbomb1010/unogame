To play, your Firebase project needs the correct security rules. Please copy the rules below and paste them into your Firestore Database 'Rules' tab.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    match /uno_games/{gameId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null &&
                    (resource.data.playerIds.hasAny([request.auth.uid]) ||
                     request.resource.data.playerIds.size() == resource.data.playerIds.size() + 1);
    }
  }
}
